    <script>
        // Set the birthday date here (Year, Month - 1, Day, Hour, Minute)
        const birthdayDate = new Date("2026-04-15T00:00:00").getTime();

        const timerInterval = setInterval(function() {
            const now = new Date().getTime();
            const distance = birthdayDate - now;

            const days = Math.floor(distance / (1000 * 60 * 60 * 24));
            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((distance % (1000 * 60)) / 1000);

            if (distance < 0) {
                clearInterval(timerInterval);
                document.getElementById("countdown-timer").innerHTML = "IT'S YOUR DAY! 🎉";
                document.getElementById("main-title").innerHTML = "Ready for your surprise?";
            } else {
                document.getElementById("countdown-timer").innerHTML = 
                    days + "d " + hours + "h " + minutes + "m " + seconds + "s left";
                document.getElementById("main-title").innerHTML = "Counting down to the big day...";
            }
        }, 1000);

        function revealWish() {
            const wish = document.getElementById('wish');
            wish.style.display = 'block';
            
            // Auto-play attempt (many browsers require user interaction first)
            const song = document.getElementById('birthday-song');
            song.play();

            window.scrollTo({
                top: wish.offsetTop - 100,
                behavior: 'smooth'
            });
        }
    </script>
